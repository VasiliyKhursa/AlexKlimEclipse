Рассмотрим несколько примеров работы с файлами PDF.

Может ли ваше устройство работать с PDF-файлами


public boolean canReadPdf(Context c) {
	PackageManager pm = c.getPackageManager();
	Intent data = new Intent(Intent.ACTION_VIEW);
	// устанавливаем MIME-тип
	data.setType("application/pdf");
	// если имеется программа для работы с PDF
	if(pm.queryIntentActivities(data, PackageManager.MATCH_DEFAULT_ONLY).size() > 0) {
		return true;
	}
	// not found any
	return false;
}

Запускаем PDF-читалку через намерение

Существует много программ, способных работать с файлами формата PDF. Для эмулятора можете скачать бесплатную читалку Android PDF Viewer или любой другой APK-файл, который сможете установить на эмулятор.

Предположим, мы знаем путь к PDF-файлу и хотим его запустить через любую читалку, установленную у пользователя.


File file = new File("/mnt/sdcard/PussyInBoots.pdf");
PackageManager packageManager = getPackageManager();
Intent intent = new Intent(Intent.ACTION_VIEW);
intent.setType("application/pdf");
List<ResolveInfo> list = packageManager.queryIntentActivities(intent,
		PackageManager.MATCH_DEFAULT_ONLY);
if (list.size() > 0 && file.isFile()) {
	intent.setAction(Intent.ACTION_VIEW);
	Uri uri = Uri.fromFile(file);
	intent.setDataAndType(uri, "application/pdf");
	startActivity(intent);
}

Сначала мы устанавливаем MIME-тип application/pdf, чтобы убедиться, что на устройстве есть приложения, способные работать с данным типом. Если такие программы обнаружатся, то запускаем намерение с указанием пути к файлу и нужным типом.

Получаем список PDF-файлов и запускаем читалку

Поняв, как можно запустить читалку, можно модифицировать программу. Создадим простой файловый менеджер, который будет выводить список PDF-файлов, находящихся в корне SD-карточки. Список будем выводить в ListView и при нажатии на элементе списка будем запускать читалку для чтения выбранного файла.


public class TestActivity extends ListActivity {

	ArrayAdapter adapter;
	ArrayList listItems = new ArrayList();
	private File[] fileslist; // массив файлов
	String[] pdflist;

	public void onCreate(Bundle savedInstanceState) {
		super.onCreate(savedInstanceState);
		setContentView(R.layout.activity_test);

		File pdf_files = Environment.getExternalStorageDirectory();
		// список файлов с расширением PDF
		fileslist = pdf_files.listFiles(new FilenameFilter() {
			public boolean accept(File dir, String name) {
				return ((name.endsWith(".pdf")));
			}
		});
		pdflist = new String[fileslist.length];
		// формируем строковый массив имен файлов
		for (int i = 0; i < fileslist.length; i++) {
			pdflist[i] = fileslist[i].getName();
		}
		this.setListAdapter(new ArrayAdapter(this,
				android.R.layout.simple_list_item_1, pdflist));
	}

	protected void onListItemClick(ListView l, View v, int position, long id) {
		super.onListItemClick(l, v, position, id);
		PackageManager packageManager = getPackageManager();
		Intent intent = new Intent(Intent.ACTION_VIEW);
		intent.setType("application/pdf");
		List<ResolveInfo> list = packageManager.queryIntentActivities(intent,
				PackageManager.MATCH_DEFAULT_ONLY);
		if (list.size() > 0 && fileslist[(int) id].isFile()) {
			intent.setAction(Intent.ACTION_VIEW);
			Uri uri = Uri.fromFile(fileslist[(int) id].getAbsoluteFile());
			intent.setDataAndType(uri, "application/pdf");
			startActivity(intent);
		}
	}
}
