В составе Android есть классы, позволяющие помещать в архив файлы, а также извлекать из архива. Этим мы сейчас и займёмся.

Архивируем в zip

Для начала создадим отдельный класс Utils, который будет содержать логику кода для архивации файла:


package ru.alexanderklimov.compresstozip;

import java.io.BufferedInputStream;
import java.io.BufferedOutputStream;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.util.zip.ZipEntry;
import java.util.zip.ZipOutputStream;

import android.util.Log;

public class Utils {
	public static final int BUFFER = 2048;

	/**
	 * to compress file
	 * 
	 * @param file
	 *            the input file to compress
	 * @param zipFile
	 *            name of the compressed file
	 * 
	 */
	public static void compress(String file, String zipFile) {
		try {
			BufferedInputStream inBufferStream = new BufferedInputStream(
					new FileInputStream(file), BUFFER);

			FileOutputStream targetOutputStream = new FileOutputStream(zipFile);
			ZipOutputStream outZipStream = new ZipOutputStream(
					new BufferedOutputStream(targetOutputStream));

			byte data[] = new byte[BUFFER];

			ZipEntry entry = new ZipEntry(
					file.substring(file.lastIndexOf("/") + 1));
			outZipStream.putNextEntry(entry);
			int count;
			while ((count = inBufferStream.read(data, 0, BUFFER)) != -1) {
				outZipStream.write(data, 0, count);
			}

			inBufferStream.close();
			outZipStream.close();

		} catch (Exception ex) {
			Log.d("Error", ex.getMessage());
		}
	}
}

Добавим на экран кнопку, чтобы по щелчку архивировать заданный файл на SD-карте в архивный файл. Напишем код:


package ru.alexanderklimov.compresstozip;

import android.os.Bundle;
import android.os.Environment;
import android.app.Activity;
import android.view.Menu;
import android.view.View;

public class MainActivity extends Activity {

	@Override
	public void onCreate(Bundle savedInstanceState) {
		super.onCreate(savedInstanceState);
		setContentView(R.layout.activity_main);
	}

	@Override
	public boolean onCreateOptionsMenu(Menu menu) {
		getMenuInflater().inflate(R.menu.activity_main, menu);
		return true;
	}

	public void onClick(View v) {
		compressToZip();
	}

	public void compressToZip() {
		// получаем путь к SD-карте
		String extStorageDirectory = Environment.getExternalStorageDirectory().toString();
		final String FILE_TARGET = extStorageDirectory + "/test.txt";
		final String FILE_ZIP = extStorageDirectory + "/test.zip";
		Utils.compress(FILE_TARGET, FILE_ZIP);
	}

}

Извлекаем из архива

Теперь рассмотрим обратную задачу и попробуем извлечь файл из архива. В класс Utils из предыдущего примера добавим ещё один метод:


/**
 * to decompress file zip
 * 
 * @param zipFile
 *            the target ZIP file to unzip
 * @param pathToUnzip
 *            where unzipped file to be stored
 * 
 */
public static void decompress(String zipFile, String pathToUnzip) {

	try {
		ZipInputStream zipIn = new ZipInputStream(new FileInputStream(
				zipFile));
		ZipEntry entry = null;
		while ((entry = zipIn.getNextEntry()) != null) {
			if (!entry.isDirectory()) {
				FileOutputStream fout = new FileOutputStream(pathToUnzip
						+ "/" + entry.getName());
				for (int c = zipIn.read(); c != -1; c = zipIn.read()) {
					fout.write(c);
				}
				zipIn.closeEntry();
				fout.close();
			}
		}

		zipIn.close();
	} catch (Exception ex) {
		Log.d("Error", ex.getMessage());
	}
}

Добавим в приложении вторую кнопку для извлечения архива и напишем код:


public void decompressFromZip() {
	// получаем путь к SD-карте
	String extStorageDirectory = Environment.getExternalStorageDirectory().toString();
	// указанная директория должна существовать!
	final String DIR_TARGET = extStorageDirectory + "/Unzip";
	final String FILE_ZIP = extStorageDirectory + "/test.zip";

	Utils.decompress(FILE_ZIP, DIR_TARGET);
}
