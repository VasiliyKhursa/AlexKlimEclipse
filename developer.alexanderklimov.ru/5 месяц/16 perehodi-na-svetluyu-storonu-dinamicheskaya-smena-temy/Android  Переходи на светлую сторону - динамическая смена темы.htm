Вся логика умещается в вызове метода setTheme(). Для выбора создадим диалоговое окно. Узнав выбор пользователя, закрываем активность и запускаем его снова с нужными настройками. Вы можете сохранить выбор пользователя в файл настроек, чтобы запоминать выбранное состояние, это вам в качестве домашнего задания.


package ru.alexanderklimov.test;

import android.app.Activity;
import android.app.AlertDialog;
import android.content.DialogInterface;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.TextView;

public class MainActivity extends Activity implements
		DialogInterface.OnClickListener {

	public final static int LIGHT_THEME = 0;
	public final static int DARK_THEME = 1;

	int mCurTheme;

	/** Called when the activity is first created. */
	@Override
	public void onCreate(Bundle savedInstanceState) {
		super.onCreate(savedInstanceState);
		
		mCurTheme = getIntent().getIntExtra("Theme", 0);
		switch (mCurTheme) {
		case LIGHT_THEME:
			setTheme(android.R.style.Theme_Holo_Light_DarkActionBar);
			break;
		case DARK_THEME:
			setTheme(android.R.style.Theme_Holo);
			break;
		default:
		}

		setContentView(R.layout.activity_test);

		TextView tvSide = (TextView) findViewById(R.id.textView1);
		if (mCurTheme == 0)
			tvSide.setText("Переходи на тёмную сторону, у нас есть печеньки!");
		else {
			tvSide.setText("Переходи на светлую сторону.");
		}

	}

	public void onClick(View v) {
		createThemeDialog();

	}

	private void createThemeDialog() {
		// текст для выбора тем
		String choose[] = { "Theme.Holo.Light.DarkActionBar (светлая)",
				"Theme.Holo (тёмная)" };

		AlertDialog.Builder builder = new AlertDialog.Builder(this);

		// Заголовок для диалогового окна
		builder.setTitle("Выбери тему");

		// Переключатели
		builder.setSingleChoiceItems(choose, 0, null);

		// Кнопка OK
		builder.setPositiveButton("OK", this);

		// Кнопка Отмена
		builder.setNegativeButton("Отмена", null);

		// Создаем окно диалога
		AlertDialog dialog = builder.create();

		// Выводим диалог
		dialog.show();
	}

	@Override
	public void onClick(DialogInterface dialog, int which) {
		// TODO Auto-generated method stub
		AlertDialog alert = (AlertDialog) dialog;
		int position = alert.getListView().getCheckedItemPosition();

		finish();
		Intent intent = new Intent(this, MainActivity.class);
		intent.putExtra("Theme", position);
		intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
		intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
		startActivity(intent);
	}
}

Внедряем возможность смены темы

После того, как вы поняли принцип смены темы, давайте напишем пример, который позволит быстро и удобно внедрять функционал смены темы в любом вашем проекте. Заодно освежим подзабытую теорию.

Для начала напомню, что описания темы хранятся в файлах styles.xml. Но это не является обязательным условием. Создадим свой файл custom_themes.xml в папке в res/values. Подобный подход позволит нам быстро скопировать готовый файл с настройками темы в любой другой проект.

Придумайте своим темам уникальные имена. Допустим, мы хотим иметь две темы - под стиль Hello Kitty (розовый) и тема чёрного кота. Это может выглядеть так:


<?xml version="1.0" encoding="utf-8"?>
<resources>

    <style name="HelloKittyTheme" parent="android:Theme.Holo.Light">
        <item name="android:background">#FF1493</item>
        <item name="android:textColor">#000000</item>
        <item name="android:actionBarStyle">@style/PinkActionBar</item>
    </style>

    <style name="BlackCatTheme" parent="android:Theme.Holo">
        <item name="android:background">#000000</item>
        <item name="android:textColor">#FFFFFF</item>
        <item name="android:actionBarStyle">@style/BlackActionBar</item>
    </style>

    <style name="PinkActionBar" parent="@android:style/Widget.Holo.Light.ActionBar">
        <item name="android:background">#FF1493</item>
    </style>

    <style name="BlackActionBar" parent="@android:style/Widget.Holo.ActionBar">
        <item name="android:background">#000000</item>
    </style>

</resources>

Здесь хотелось бы обратить внимание на отдельные стили для ActionBar. Без них у меня в примерах появлялись артефакты - незакрашенные прямоугольники в заголовке приложения. Если вы пишите программы для старых устройств до API 11, то можете обойтись без дополнительных стилей.

Займемся кодом. Вынесем логику смены темы в отдельный класс-обёртку. Вы сможете потом просто копировать созданный класс в другой проект и быстро настроить необходимые параметры, например, добавить новые стили. Класс выглядит следующим образом:

ThemeUtils.java


package ru.alexanderklimov.choosetheme;

import android.app.Activity;
import android.content.Intent;

public class ThemeUtils {

	private static int cTheme;
	public final static int BLACK = 0;
	public final static int PINK = 1;

	public static void changeToTheme(Activity activity, int theme) {
		cTheme = theme;

		activity.finish();
		activity.startActivity(new Intent(activity, activity.getClass()));
	}

	public static void onActivityCreateSetTheme(Activity activity, int theme) {
		switch (cTheme) {
		default:
		case BLACK:
			activity.setTheme(R.style.BlackCatTheme);
			break;
		case PINK:
			activity.setTheme(R.style.HelloKittyTheme);
			break;
		}
	}
}

Чтобы подключить новую тему, достаточно завести новую константу и добавить код в блок switch. В данном примере темой по умолчанию считается чёрная тема.

Осталось написать код для активности. Для удобства я решил сохранять выбранные настройки в файл настроек, чтобы при следующем запуске автоматически загружалась нужная тема.


package ru.alexanderklimov.choosetheme;

import android.os.Bundle;
import android.app.Activity;
import android.content.Context;
import android.content.SharedPreferences;
import android.content.SharedPreferences.Editor;
import android.view.Menu;
import android.view.View;
import android.widget.EditText;

public class MainActivity extends Activity {
	
	public static final String APP_PREFERENCES = "mysettings"; // это будет именем файла настроек
	public static final String APP_PREFERENCES_THEME = "Theme";
	SharedPreferences mSettings;
	

	@Override
	protected void onCreate(Bundle savedInstanceState) {
		super.onCreate(savedInstanceState);
		
		mSettings = getSharedPreferences(APP_PREFERENCES, Context.MODE_PRIVATE);
		
		if(mSettings.contains(APP_PREFERENCES_THEME)) {
			ThemeUtils.onActivityCreateSetTheme(this, mSettings.getInt(APP_PREFERENCES_THEME, 0));
		}
		
		setContentView(R.layout.activity_main);
	}

	@Override
	public boolean onCreateOptionsMenu(Menu menu) {
		// Inflate the menu; this adds items to the action bar if it is present.
		getMenuInflater().inflate(R.menu.main, menu);
		return true;
	}
	
	public void onClick(View v){
		Editor editor = mSettings.edit();
		
		switch (v.getId()) {
		case R.id.pinkbutton:
			editor.putInt(APP_PREFERENCES_THEME, ThemeUtils.PINK);
			editor.commit();
			ThemeUtils.changeToTheme(this, ThemeUtils.PINK);
			break;
		case R.id.blackbutton:
			editor.putInt(APP_PREFERENCES_THEME, ThemeUtils.BLACK);
			editor.commit();
			ThemeUtils.changeToTheme(this, ThemeUtils.BLACK);
			break;
		default:
			break;
		}
	}
}
